Horizontal

Code
<div class="row">
<div class="column desktop-7 tablet-12">
<div class="swiper" id="swiperDefault">
<a class="arrow-left" href="#"><i class="icon chevron circle left"></i></a>
<a class="arrow-right" href="#"><i class="icon chevron circle right"></i></a>
<div class="swiper-container">
<div class="swiper-wrapper">
<div class="swiper-slide"> <img src="images/banners/banner-1.jpg" width="100%" height="auto"> </div>
<div class="swiper-slide"> <img src="images/banners/banner-2.jpg" width="100%" height="auto"> </div>
<div class="swiper-slide"> <img src="images/banners/banner-3.jpg" width="100%" height="auto"> </div>
<div class="swiper-slide"> <img src="images/banners/banner-4.jpg" width="100%" height="auto"> </div>
<div class="swiper-slide"> <img src="images/banners/banner-5.jpg" width="100%" height="auto"> </div>
</div>
</div>
<div class="pagination"></div>
</div>
</div>
</div>

<script>
var mySwiperDefault = $('#swiperDefault .swiper-container').swiper({
pagination: '#swiperDefault .pagination',
loop: true,
grabCursor: true,
paginationClickable: true,
calculateHeight: true
});
$('#swiperDefault .arrow-left').on('click', function(e){
e.preventDefault()
mySwiperDefault.swipePrev()
})
$('#swiperDefault .arrow-right').on('click', function(e){
e.preventDefault()
mySwiperDefault.swipeNext()
})
</script>
